      function createCookies() {
          //set cookie expiration date variable
          var d = new Date();
          var expiration_days = 30;
          d.setTime(d.getTime() + (expiration_days * 24 * 60 * 60 * 1000));
          var expires = "expires=" + d.toGMTString();

          //set all the other cookie variables
          var cookie1 = "cookie1";
          var cookie1_value = document.getElementById("input_cookie1").value;
          var cookie2 = "cookie2";
          var cookie2_value = document.getElementById("input_cookie2").value;
          var cookie3 = "cookie3";
          var cookie3_value = document.getElementById("input_cookie3").value;

          setCookie(cookie1, cookie1_value, expires);
          setCookie(cookie2, cookie2_value, expires);
          setCookie(cookie3, cookie3_value, expires);

          //notify user that the cookies were created
          document.getElementById("show_cookie_creation_validation").innerHTML = "document.cookie has been modified";

          cookieCheck();
      }

      function setCookie(cookie_name, cookie_value, expires) {
          if (cookie_value !== "" && cookie_value !== null) {
              document.cookie = cookie_name + "=" + cookie_value + "; " + expires;
          } else {
              //ensure that a key is not created without a value 
              deleteCookie(cookie_name);
          }
      }

      function deleteCookie(cookie_name) {
          document.cookie = cookie_name + "=; expires=Thu, 01 Jan 1970 00:00:00 UTC";
          document.getElementById("show_cookie_deleted").innerHTML = "A cookie has been deleted!";
      }

      function getCookie(cookie_name) {
          var name = cookie_name + "=";
          var cookie_pair = document.cookie.split(';');

          for (var i = 0; i < cookie_pair.length; i++) {
              var cookie_pair_string = cookie_pair[i];

              while (cookie_pair_string.charAt(0) == ' ') {
                  cookie_pair_string = cookie_pair_string.substring(1);
              }
              
              if (cookie_pair_string.indexOf(name) === 0) {
                  return cookie_pair_string.substring(name.length, cookie_pair_string.length);
              }
          }

          return "";
      }
